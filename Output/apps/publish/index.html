Static Site Generation with Swift and Publish

// Swift sample post
   static func swiftItem<T: Website>(for item: Item<T>) -> Node {
       print(item.title)
       return .div(
           .class("content"),
           .id(item.title),
           .contentBody(item.body),
           .a(.href("#\(item.title)"),
           .text("Permalink"))
       )
   }
